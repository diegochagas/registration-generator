for(let i = 1; i <= 8; i++){

digits[i] = generateRandomNumbers(n) % 9;

function generateRazaoSocial (name, familyName){

const type = ['MEI', 'ME', 'EPP', 'EI', 'EIRELI', 'LTDA', 'SA'];
var i = generateRandomNumbers(type.length - 1);
return companyName + " " + segment + " " + type[i];
}
